// 팝업 열기 함수에 터치 강제 방지 해제 추가 function openAssetPopup(assetType) { // ... 기존 데이터 처리 로직 ... const pop = document.getElementById("popOverlay"); pop.classList.add("open"); // 아이폰에서 배경 스크롤 방지 document.documentElement.style.overflow = "hidden"; document.body.style.overflow = "hidden"; } // 팝업 닫기 함수 수정 function closePopup(e) { // 배경을 누르거나 X를 눌렀을 때만 닫기 if (e && e.target !== document.getElementById("popOverlay") && e.target !== document.querySelector(".popup-close")) return; document.getElementById("popOverlay").classList.remove("open"); document.documentElement.style.overflow = ""; document.body.style.overflow = ""; } // 스와이프 감도 조절 (아이폰 최적화) let touchStartY = 0; const sheet = document.getElementById("popSheet"); sheet.addEventListener("touchstart", e => { touchStartY = e.touches[0].clientY; }, { passive: true }); sheet.addEventListener("touchend", e => { const touchEndY = e.changedTouches[0].clientY; // 아래로 100px 이상 빠르게 쓸어내릴 때만 닫기 if (touchEndY - touchStartY > 100) { closePopup(); } }, { passive: true });
